def make_readable(seconds): 
    
    secon = [seconds // 3600, seconds % 3600 // 60, seconds % 60] 
    
    times = ['00', ':', '00', ':', '00'] 
    
    for i in range(3): 
    
        if secon[i] > 9: 
    
            times[i * 2] = str(secon[i]) 
    
        elif 0 != secon[i] < 10: 
    
            times[i * 2] = '0' + str(secon[i]) 
    
    return ''.join(times)
